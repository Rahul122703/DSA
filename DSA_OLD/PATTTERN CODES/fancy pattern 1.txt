#include <iostream>
using namespace std;

int main(){
    int n = 5;
    int start = 8,end = 8;
    for(int row = 1 ; row <= n ; row++){
        
        for(int col = 0 ; col < 17; col++){
            int condition2 = (start == col || end == col || (col<end && col >start));
            if(row%2){
                if (!(col%2) && condition2){
                    cout<<row;
                }
                else{
                    cout<<"*";
                }
            }
            else if (!(row%2)){
                if(col%2 && condition2){
                    cout<<row;
                }
                else{
                    cout<<"*";
                }
            }
        }
        start = start - 1;
        end = end + 1;
        
        cout<<endl;
    }
}
